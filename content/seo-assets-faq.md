# FAQPage JSON-LD шаблоны для сервисных страниц

> Файл подготовлен SEO Specialist (CMPA-59).
> Данные FAQ берутся из `content/data/faq.json`.
> Каждый блок вставляется в `<script type="application/ld+json">` на соответствующей странице.
> Домен-плейсхолдер: `https://ruslanweb.vercel.app`

---

## Как внедрить в Next.js

```tsx
// Пример компонента для FAQPage JSON-LD
// app/services/[slug]/page.tsx

import faqData from '@/content/data/faq.json'

function FAQSchema({ slug }: { slug: string }) {
  const faqs = faqData[slug as keyof typeof faqData]
  if (!faqs) return null

  const schema = {
    "@context": "https://schema.org",
    "@type": "FAQPage",
    "mainEntity": faqs.map(({ question, answer }) => ({
      "@type": "Question",
      "name": question,
      "acceptedAnswer": {
        "@type": "Answer",
        "text": answer
      }
    }))
  }

  return (
    <script
      type="application/ld+json"
      dangerouslySetInnerHTML={{ __html: JSON.stringify(schema) }}
    />
  )
}
```
